# Todos App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `todos.html`, a companion todos app sharing the same JSONBin bin as `kanban.html`, with three fixed sections (Today / This Week / Someday) and the ability to promote a todo into a kanban card.

**Architecture:** Clone `kanban.html` as the base, stripping board/modal/drag/label logic. Extract four pure state-mutation functions into `todos-state.js` so they can be tested independently. `todos.html` includes `todos-state.js` and adds section layout, checkbox toggle, inline add, section re-file, and promote interactions.

**Tech Stack:** Vanilla JS, no framework, no build step. JSONBin.io REST API. Browser-runnable tests in `tests/todos.test.html`.

```js
// Pure state mutation functions for todos.html.
// Each function returns a new state object — never mutates in place.

function toggleDone(state, section, id) {
  return {
    ...state,
    todos: {
      ...state.todos,
      [section]: (state.todos[section] || []).map(item =>
        item.id === id ? { ...item, done: !item.done } : item
      ),
    },
  };
}

function clearDone(state) {
  const sections = ['today', 'thisWeek', 'someday'];
  const cleaned = {};
  for (const s of sections) cleaned[s] = (state.todos[s] || []).filter(i => !i.done);
  return { ...state, todos: cleaned };
}

function moveItem(state, fromSection, toSection, id) {
  if (fromSection === toSection) return state;
  const item = (state.todos[fromSection] || []).find(i => i.id === id);
  if (!item) return state;
  return {
    ...state,
    todos: {
      ...state.todos,
      [fromSection]: state.todos[fromSection].filter(i => i.id !== id),
      [toSection]:   [...(state.todos[toSection] || []), item],
    },
  };
}

function promoteToKanban(state, section, id) {
  const item = (state.todos[section] || []).find(i => i.id === id);
  if (!item) return state;
  return {
    ...state,
    todo: [
      ...(state.todo || []),
      { id: crypto.randomUUID(), title: item.title, body: '', tag: '', tagClass: '' },
    ],
    todos: {
      ...state.todos,
      [section]: state.todos[section].filter(i => i.id !== id),
    },
  };
}
```

```html
<script>
// ─── Config ───────────────────────────────────────────────────────────────────
const API     = 'https://api.jsonbin.io/v3';
const POLL_MS = 15_000;

const SECTIONS = [
  { id: 'today',    label: 'Today'     },
  { id: 'thisWeek', label: 'This Week' },
  { id: 'someday',  label: 'Someday'   },
];

// ─── Config from URL / localStorage ──────────────────────────────────────────
let MASTER_KEY = null;
let binId      = null;

(function parseConfig() {
  const raw = location.hash.slice(1) || localStorage.getItem('kanban-config');
  if (!raw) return;
  const slash = raw.lastIndexOf('/');
  if (slash !== -1) {
    MASTER_KEY = raw.slice(0, slash) || null;
    binId      = raw.slice(slash + 1) || null;
  } else {
    binId = raw;
  }
})();

function saveConfig() {
  const cfg = MASTER_KEY ? `${MASTER_KEY}/${binId || ''}` : (binId || '');
  history.replaceState(null, '', '#' + cfg);
  localStorage.setItem('kanban-config', cfg);
}

// ─── State ────────────────────────────────────────────────────────────────────
let state     = { todo: [], inprogress: [], done: [], todos: { today: [], thisWeek: [], someday: [] } };
let saveTimer = null;
let pollTimer = null;
let isSaving  = false;

// ─── Status ───────────────────────────────────────────────────────────────────
function setStatus(kind, msg) {
  document.getElementById('status-dot').className    = kind;
  document.getElementById('status-text').textContent = msg;
}

// ─── JSONBin API ──────────────────────────────────────────────────────────────
const headers = () => ({ 'X-Master-Key': MASTER_KEY });

async function createBin() {
  const res = await fetch(`${API}/b`, {
    method: 'POST',
    headers: { ...headers(), 'Content-Type': 'application/json', 'X-Bin-Name': 'kanban-shared', 'X-Bin-Private': 'false' },
    body: JSON.stringify({ todo: [], inprogress: [], done: [], todos: { today: [], thisWeek: [], someday: [] } }),
  });
  if (!res.ok) throw new Error(`Create bin failed: ${res.status}`);
  const { metadata } = await res.json();
  return metadata.id;
}

async function fetchState() {
  const res = await fetch(`${API}/b/${binId}/latest`, { headers: headers() });
  if (res.status === 404) throw Object.assign(new Error('Bin not found'), { notFound: true });
  if (!res.ok) throw new Error(`Fetch failed: ${res.status}`);
  const { record } = await res.json();
  // Migrate: bin created by kanban.html may not have todos key
  if (!record.todos) record.todos = { today: [], thisWeek: [], someday: [] };
  return record;
}

async function pushState() {
  const res = await fetch(`${API}/b/${binId}`, {
    method: 'PUT',
    headers: { ...headers(), 'Content-Type': 'application/json' },
    body: JSON.stringify(state),
  });
  if (!res.ok) throw new Error(`Save failed: ${res.status}`);
}

// ─── Debounced save ───────────────────────────────────────────────────────────
function scheduleSave() {
  setStatus('saving', 'Saving…');
  clearTimeout(saveTimer);
  clearTimeout(pollTimer);
  saveTimer = setTimeout(async () => {
    isSaving = true;
    try {
      await pushState();
      setStatus('saved', 'Saved');
    } catch (e) {
      setStatus('error', 'Save failed');
      console.error(e);
    } finally {
      isSaving = false;
      schedulePoll();
    }
  }, 700);
}

// ─── Polling ──────────────────────────────────────────────────────────────────
function schedulePoll() {
  clearTimeout(pollTimer);
  pollTimer = setTimeout(async () => {
    if (isSaving || saveTimer) return;
    try {
      const remote = await fetchState();
      state = remote;
      renderBoard();
      setStatus('saved', 'Synced');
    } catch (_) {}
    schedulePoll();
```

```js
// ─── Render ───────────────────────────────────────────────────────────────────
function renderBoard() {
  const board = document.getElementById('board');
  board.innerHTML = '';

  for (const sec of SECTIONS) {
    const items = (state.todos && state.todos[sec.id]) || [];
    const active = items.filter(i => !i.done).length;

    const el = document.createElement('div');
    el.className = `section section-${sec.id.toLowerCase()}`;
    el.dataset.section = sec.id;

    el.innerHTML = `
      <div class="section-header">
        <span class="section-title">${sec.label}</span>
        <span class="section-count">${active || ''}</span>
      </div>
      <div class="todo-list"></div>
      <div class="todo-add-row">
        <input class="todo-add-input" placeholder="+ add item" type="text" autocomplete="off" data-section="${sec.id}">
      </div>
    `;

    const list = el.querySelector('.todo-list');
    for (const item of items) list.appendChild(buildItem(item, sec.id));

    board.appendChild(el);
    wireAddInput(el.querySelector('.todo-add-input'), sec.id);
  }
}

function buildItem(item, sectionId) {
  const div = document.createElement('div');
  div.className = 'todo-item';
  div.dataset.id = item.id;

  div.innerHTML = `
    <div class="todo-checkbox${item.done ? ' checked' : ''}"></div>
    <span class="todo-title${item.done ? ' done' : ''}">${esc(item.title)}</span>
    <div class="todo-actions">
      <div class="section-picker">
        <button class="todo-action-btn btn-move" title="Move to section">⇄</button>
        <div class="section-picker-menu">
          ${SECTIONS.map(s => `
            <button class="picker-option" data-target="${s.id}"
              ${s.id === sectionId ? 'disabled' : ''}>${s.label}</button>
          `).join('')}
        </div>
      </div>
      <button class="todo-action-btn btn-promote" title="Send to kanban board">→ board</button>
    </div>
  `;

  // Checkbox toggle
  div.querySelector('.todo-checkbox').addEventListener('click', () => {
    state = toggleDone(state, sectionId, item.id);
    renderBoard();
    scheduleSave();
  });

  // Section picker toggle
  const moveBtn  = div.querySelector('.btn-move');
  const menu     = div.querySelector('.section-picker-menu');
  moveBtn.addEventListener('click', e => {
    e.stopPropagation();
    menu.classList.toggle('open');
    if (menu.classList.contains('open')) {
      const close = () => { menu.classList.remove('open'); document.removeEventListener('click', close); };
      setTimeout(() => document.addEventListener('click', close), 0);
    }
  });

  div.querySelectorAll('.picker-option:not([disabled])').forEach(btn => {
    btn.addEventListener('click', e => {
      e.stopPropagation();
      state = moveItem(state, sectionId, btn.dataset.target, item.id);
      renderBoard();
      scheduleSave();
    });
  });

  // Promote to kanban
  div.querySelector('.btn-promote').addEventListener('click', () => {
    state = promoteToKanban(state, sectionId, item.id);
    renderBoard();
    scheduleSave();
  });

  return div;
}

function wireAddInput(input, sectionId) {
  input.addEventListener('keydown', e => {
    if (e.key === 'Enter') {
      const title = input.value.trim();
      if (!title) return;
      state.todos[sectionId].push({ id: crypto.randomUUID(), title, done: false });
      renderBoard();
      scheduleSave();
    }
    if (e.key === 'Escape') {
      input.value = '';
      input.blur();
    }
  });
}
```
